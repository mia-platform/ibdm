package catalog

import (
	"context"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"net/url"
	"strings"
	"sync"
	"time"

	"github.com/google/uuid"
	"github.com/lestrrat-go/jwx/v3/jwa"
	"github.com/lestrrat-go/jwx/v3/jwt"
)

const (
	clientAssertionType = "urn:ietf:params:oauth:client-assertion-type:jwt-bearer"
	assertionTTL        = 5 * time.Minute
	tokenExpiryBuffer   = 30 * time.Second
)

// cachedToken holds a single realm's access token and when it expires locally.
type cachedToken struct {
	accessToken string
	expiresAt   time.Time
}

// tokenProvider fetches and caches Keycloak access tokens per realm using
// the private_key_jwt client authentication method (RFC 7523).
type tokenProvider struct {
	cfg    *config.ClientConfig
	keys   *JWKKeys
	cache  map[string]*cachedToken
	mu     sync.Mutex
	client *http.Client
}

func newTokenProvider(cfg *config.ClientConfig, keys *jwk.Keys) *tokenProvider {
	return &tokenProvider{
		cfg:    cfg,
		keys:   keys,
		cache:  make(map[string]*cachedToken),
		client: &http.Client{Timeout: 10 * time.Second},
	}
}

// Get returns a valid access token for the given realm.
// It returns a cached token if one exists and is not close to expiry; otherwise
// it fetches a fresh token from Keycloak and updates the cache.
func (t *tokenProvider) Get(ctx context.Context, realm string) (string, error) {
	t.mu.Lock()
	defer t.mu.Unlock()

	if cached, ok := t.cache[realm]; ok && time.Now().Before(cached.expiresAt) {
		return cached.accessToken, nil
	}

	tokenEndpoint := t.tokenEndpointURL(realm)
	accessToken, expiresIn, err := t.fetchToken(ctx, tokenEndpoint)
	if err != nil {
		return "", fmt.Errorf("keycloak token: realm %q: %w", realm, err)
	}

	expiry := time.Now().Add(time.Duration(expiresIn)*time.Second - tokenExpiryBuffer)
	t.cache[realm] = &cachedToken{accessToken: accessToken, expiresAt: expiry}
	return accessToken, nil
}

// tokenEndpointURL derives the Keycloak token endpoint URL for the given realm.
func (t *tokenProvider) tokenEndpointURL(realm string) string {
	base := strings.TrimRight(t.cfg.RegistrationServerBaseURL, "/")
	return fmt.Sprintf("%s/realms/%s/protocol/openid-connect/token", base, realm)
}

// fetchToken performs the token endpoint POST and returns the access token and
// its server-reported lifetime in seconds.
func (t *tokenProvider) fetchToken(ctx context.Context, tokenEndpoint string) (accessToken string, expiresIn int, err error) {
	assertion, err := t.buildAssertion(tokenEndpoint)
	if err != nil {
		return "", 0, fmt.Errorf("build JWT assertion: %w", err)
	}

	form := url.Values{}
	form.Set("grant_type", "client_credentials")
	form.Set("client_id", t.cfg.ClientID)
	form.Set("client_assertion_type", clientAssertionType)
	form.Set("client_assertion", string(assertion))
	if t.cfg.Scope != "" {
		form.Set("scope", t.cfg.Scope)
	}
	for k, v := range t.cfg.Extra {
		form.Set(k, v)
	}

	req, err := http.NewRequestWithContext(ctx, http.MethodPost, tokenEndpoint, strings.NewReader(form.Encode()))
	if err != nil {
		return "", 0, fmt.Errorf("build token request: %w", err)
	}
	req.Header.Set("Content-Type", "application/x-www-form-urlencoded")

	resp, err := t.client.Do(req)
	if err != nil {
		return "", 0, fmt.Errorf("POST %s: %w", tokenEndpoint, err)
	}
	defer resp.Body.Close()

	body, _ := io.ReadAll(resp.Body)
	if resp.StatusCode != http.StatusOK {
		return "", 0, fmt.Errorf("token endpoint returned HTTP %d: %s", resp.StatusCode, body)
	}

	var tokenResp struct {
		AccessToken string `json:"access_token"`
		ExpiresIn   int    `json:"expires_in"`
	}
	if err := json.Unmarshal(body, &tokenResp); err != nil {
		return "", 0, fmt.Errorf("parse token response: %w", err)
	}
	if tokenResp.AccessToken == "" {
		return "", 0, fmt.Errorf("token response missing access_token field")
	}

	return tokenResp.AccessToken, tokenResp.ExpiresIn, nil
}

// buildAssertion creates and signs a JWT client assertion for the given token endpoint.
// The assertion conforms to RFC 7523 §2.2 and is signed with the configured algorithm.
func (t *tokenProvider) buildAssertion(tokenEndpoint string) ([]byte, error) {
	algStr := t.cfg.PreferredTokenEndpointSignAlg
	alg, ok := jwa.LookupSignatureAlgorithm(algStr)
	if !ok {
		return nil, fmt.Errorf("unsupported signing algorithm %q", algStr)
	}

	now := time.Now()
	tok, err := jwt.NewBuilder().
		Issuer(t.cfg.ClientID).
		Subject(t.cfg.ClientID).
		Audience([]string{tokenEndpoint}).
		IssuedAt(now).
		Expiration(now.Add(assertionTTL)).
		JwtID(uuid.NewString()).
		Build()
	if err != nil {
		return nil, fmt.Errorf("build JWT claims: %w", err)
	}

	signed, err := jwt.Sign(tok, jwt.WithKey(alg, t.keys.PrivateKey))
	if err != nil {
		return nil, fmt.Errorf("sign JWT assertion: %w", err)
	}

	return signed, nil
}
